#
# Creating Mathematics function Solving python class object
# you can use as you want
#
class Math:
    def __init__(self, x,y):
        self.x = x
        self.y = y
        
    # Adding Values
    def add(self):
        return self.x + self.y
    
    # substracting values
    def substract(self):
        return self.x - self.y
    
    # multiplying value
    def multiply(self):
        return self.x * self.y
    
    # dividing value
    def divide(self):
        return self.x / self.y
    
    # square value
    def square(self):
        return self.x ** 2
