2. Triangle Herons Formula Function

// Author : Suraj Singh Bisht
// Email : surajsinghbisht054@gmail.com
// Code : C
// 
//  To Perform Triangle Herons Formula
//
#include <stdio.h>
#include <math.h>


int main(){
 float a,b,c,d,s, area;
 printf("[+] Enter Sides Of Triangles : A B C  ");
 scanf("%f %f %f", &a, &b, &c);
 s = (a+b+c)/2;
 area = sqrt(s*(s-a)*(s-b)*(s-c));
 printf("[+] Area Of Triangle is : %f\n", area);
}

6. Cubelist Function

// ========================================================================
// -------------------x cubelist.cpp x-------------------------------------
// ------------------------------------------------------------------------
// demonstrating For loop with setw() functions
// Author : Suraj Singh Bisht
// Email : surajsinghbisht054@gmail.com
// Code : C++


#include <iostream>
#include <iomanip>
using namespace std;

int main(){
 unsigned long times;
 long cube;
 cout << "\n [+] Cube list From 1 to : ";
 cin >> times;
 cout <<"\n\n|----------------------------|"<< endl;
 for(int i=0; i<=times; i++){
  cube=i*i*i;
  cout << " | "<< setw(6) << i << " | "<<setw(15) << cube << " | "<< endl;
 }
}
