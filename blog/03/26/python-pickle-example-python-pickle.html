Input : [20]  

#
# Let's try something more interesting
# Here i am creating a class that can multiply
# two numbers provide by the user
#
class multiply:
    
    def __init__(self, x,y):
        self.x = x
        self.y = y
        
    def get(self):
        return self.x * self.y

Output : [20]
