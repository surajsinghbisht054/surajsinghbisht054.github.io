# Creating Class
class Number_Convert:
 def __init__(self, num=None):
  self.num=num
 def convert(self):
  raw=self.cutt()
  raw1=''
  for i in raw.split(','):
   raw1=raw1+self.tranl(i)+','

  raw1=raw1.split(',')
  raw1.reverse()
  st=[]
  for i in range(len(raw1)):
   if len(raw1[i])!=0:
    data=str(raw1[i])+str(val[i])
    st.append(data)
   else:
    pass
  st.reverse()
  data=''
  for i in st:
   data=data+i
  return data

 def cutt(self):
  data=self.num[::-1]
  k=0
  l=''
  for i in data:
   if k%2==1 and k!=1:
    l=l+','+i
    k=k+1
    pass
   else:
    l=l+i
    k=k+1
    pass
  data=l[::-1]
  return data
 def tranl(self, i):
  data=i
  if int(data)==0:
   return '' 
  elif int(data)<=20:

   rw=form[x]+'-'+form[y]
   return rw
