import urllib2
import sys
from HTMLParser import HTMLParser


# For More Info https://docs.python.org/2/library/htmlparser.html
class link_extractor(HTMLParser):
    def handle_starttag(self,tag, attrs):
        for attr in attrs:
            if 'href' in attr[0]:
                print attr[1]

  page=None
 return page
